#normfunmu: sigma:pdf:np.exp():
def normfun(x, mu, sigma):
    pdf = np.exp(-((x - mu)**2) / (2* sigma**2)) / (sigma * np.sqrt(2*np.pi))
    return pdf

def norm_generate(dataset):
    x = np.arange(dataset.min(), dataset.max(),1)
    y = normfun(x, dataset.mean(), dataset.std())
    return x, y
